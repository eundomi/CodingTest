const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout,
});

  input.push(x);
});
rl.on("close", function () {
  let arr = input[0].split(" ").map(Number);
  solution(arr);
});

function solution(arr) {
  let max = arr[0];
  let sum = arr[0] + arr[1] + arr[2];

  if (max < arr[1]) {
    max = arr[1];
  }
  if (max < arr[2]) {
    max = arr[2];
  }

  if (sum - max <= max) {
    console.log("NO");
    return;
  }
  console.log("YES");
}
